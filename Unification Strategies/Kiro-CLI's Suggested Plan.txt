   569+                  df = df_data.get(term, 0)
   570+                  idf = math.log((corpus_stats.n_chunks - df + 0.5) / (df + 0.5) + 1.0)
   571+
   572+                  # TF component
   573+                  tf = chunk_tf.get(term, 0)
   574+                  tf_component = (tf * (self.K1 + 1)) / (
   575+                      tf + self.K1 * (1 - self.B + self.B * chunk.token_count / corpus_stats.avgdl)
   576+                  )
   577+
   578+                  # Synonym boost (from all implementations)
   579+                  boost = self.SYNONYM_BOOST if term in parsed.synonym_terms else 1.0
   580+
   581+                  score += idf * tf_component * boost
   582+
   583+              results.append(ScoredResult(chunk=chunk, score=score))
   584+
   585+          # Min-max normalization (from Lovable + Claude)
   586+          return self._normalize_scores(results)
   587+
   588+      def _normalize_scores(self, results: List[ScoredResult]) -> List[ScoredResult]:
   589+          """Normalize scores to [0, 1] range"""
   590+          if not results:
   591+              return results
   592+
   593+          scores = [r.score for r in results]
   594+          min_score = min(scores)
   595+          max_score = max(scores)
   596+
   597+          if max_score == min_score:
   598+              for r in results:
   599+                  r.normalized_score = 1.0
   600+          else:
   601+              for r in results:
   602+                  r.normalized_score = (r.score - min_score) / (max_score - min_score)
   603+
   604+          return sorted(results, key=lambda r: r.normalized_score, reverse=True)
   605+  ```
